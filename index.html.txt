        @keyframes fly {

<body>

    <div class="header">AQUA COINS</div>
    <div class="score-container" id="score">0</div>

    <div class="click-area">
        <div class="drop" id="dropButton"></div>
    </div>

    <script>
        let tg = window.Telegram.WebApp;
        tg.expand(); // Розгортаємо гру на повний екран

        let score = 0;
        const scoreEl = document.getElementById('score');
        const dropBtn = document.getElementById('dropButton');

        dropBtn.addEventListener('click', (e) => {
            score++;
            scoreEl.innerText = score;

            // Створюємо ефект бризок води
            for (let i = 0; i < 6; i++) {
                createSplash(e.clientX, e.clientY);
            }
            
            // Вібрація на смартфонах (якщо підтримується)
            if(tg.HapticFeedback) {
                tg.HapticFeedback.impactOccurred('medium');
            }
        });

        function createSplash(clickX, clickY) {
            const splash = document.createElement('div');
            splash.classList.add('splash');
            
            const rect = dropBtn.getBoundingClientRect();
            const x = clickX - rect.left - (rect.width / 2);
            const y = clickY - rect.top - (rect.height / 2);
            
            splash.style.left = `calc(50% + ${x}px)`;
            splash.style.top = `calc(50% + ${y}px)`;
            
            // Випадковий напрямок розлітання бризок
            const angle = Math.random() * Math.PI * 2;
            const distance = 50 + Math.random() * 80;
            splash.style.setProperty('--x', `${Math.cos(angle) * distance}px`);
            splash.style.setProperty('--y', `${Math.sin(angle) * distance}px`);
            
            dropBtn.parentElement.appendChild(splash);
            setTimeout(() => splash.remove(), 400);
        }
    </script>
